#include <iostream> 
using namespace std;
int main() {
	setlocale(LC_ALL, "RUS");
	cout << "¬ведите диапазон" << endl;
	
	int a, b, c, k;
	cin >> a >> b;
	c = a;
	k = a;
	
	// for
	for (int i = a; i <= b; i++) {
		if (i % 3 == 0) {
			cout << i << " ";
		}
	}
	cout << endl;

	// while
	while (c <= b) {
		if (c % 3 == 0) {
			cout << c << " ";
		}
		c += 1;
	}
	cout << endl;
	
	// do
	do {
		if (k % 3 == 0) {
			cout << k << " ";
		}
		k += 1;
	} while (k <= b);
	cout << endl;
	
	system("pause");
	return 0;
}

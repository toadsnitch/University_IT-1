#include <iostream> 
#include <cmath>
using namespace std;
int main() {
	setlocale(LC_ALL, "RU");
	float x, y;
	cout << "Введите координаты точки" << endl;
	cin >> x >> y;

	if (y > 0 && (x*x + y*y < 9)) {
		cout << "Да\n";
	}
	else if ((y == 0 && fabs(x) <= 3) || (y >= 0 && (x * x + y * y == 9))) {
		cout << "На границе\n";
	}
	else {
		cout << "Нет\n";
	}
	system("pause");
	return 0;
}

#include <iostream> 
using namespace std;
int rp(int a) {
    return ((a % 10) * 10) + (a / 10);
}

int main() {
    setlocale(LC_ALL, "RUS");

    cout << "Введите число\n";
    int x, t_m, count = 0;
    cin >> x;
    t_m = x;

    while (t_m != 0) {
        count += 1;
        t_m /= 10;
    }


    if (count == 2) {
        cout << rp(x) << endl;
    }
    else {
        cout << "Введено не двухзначное число\n\n";
    }
    system("pause");
    return 0;
}

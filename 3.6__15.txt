#include <iostream> 
#include <cmath>
using namespace std;

int fc(double x) {
	if ((x - 1) < 1) {
		return 1;
	}
	else if ((x - 1) == 1) {
		return 0;
	}
	else if ((x - 1) > 1) {
		return -1;
	}
}

int main() {
	setlocale(LC_ALL, "RUS");
	cout << "Введите диапазон [a, b]\n";
	double a, b, h;
	cin >> a >> b;
	cout << "Введите шаг\n";
	cin >> h;
	cout << endl;

	if ((a > b && h < 0) || (a < b && h > 0)) {
		for (a; a <= b; a += h) {
			cout << a << "\t" << fc(a) << endl;
		}
	}
	else {
		cout << "Данные введены некорректно\n";
	}

	system("pause");
	return 0;
}

#include <iostream>
# include <iostream>
using namespace std;
int main() {
	int n, temporary;
	cout << "enter n\n";
	cin >> n;
	double x = 0;
	temporary = 1;
	
	for (int i = 1; i <= n; i++) {
		x += pow(i, 3) * temporary;
		temporary *= -1;
	}
	
	cout << "The amount = ";
	cout << x << endl;
	
	system("pause");
	return 0;
}

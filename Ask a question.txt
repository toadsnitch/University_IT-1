#include <iostream> 
using namespace std;
int main() {
    setlocale(LC_ALL, "RUS");
    bool flag = true;
    int count_1 = 1, count_2 = 9;
    
    for (int i = 1; i <= 8; i++) {

        if (flag == true) {
            for (int a1 = 1; a1 <= count_1; a1++) {
                cout << a1 << ' ';
            }
            count_1 += 1;
            flag = false;
            cout << endl;
        }

        else {
            for (int a9 = 9; a9 >= count_2; a9 =- 2) {
                cout << a9 << ' ';
            }
            count_2 -= 2;
            flag = true;
            cout << endl;
        }
    }
}

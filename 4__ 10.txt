#include <iostream>
using namespace std;
int main() {
	int b1 = 2, b2 = 4, b, n;
	cout << "n = ";
	cin >> n;
	cout << "b1: " << b1 << endl << "b2: " << b2 << endl;
	
	for (int i = 3; i <= n; i++) {
		b = 6 * b2 - b1;
		cout << "b" << i << ": " << b << endl;
		b1 = b2;
		b2 = b;
	}

	system("pause");
	return 0;
}
